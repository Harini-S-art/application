import requests
import tkinter as tk
from tkinter import messagebox

def get_weather():
    city = city_entry.get()
    api_key = "8fed639ec4f2fdff6fdbfcd660e57c92"
    url = f"https://api.openweathermap.org/data/2.5/weather?q={city}&appid={api_key}&units=metric"
    
    response = requests.get(url)
    data = response.json()
    
    if data.get("cod") != 200:
        messagebox.showerror("Error", "City not found!")
        return

    temp = data["main"]["temp"]
    humidity = data["main"]["humidity"]
    desc = data["weather"][0]["description"]

    result_label.config(text=f"Temp: {temp}Â°C\nHumidity: {humidity}%\nCondition: {desc.title()}")

GUI Setup
root = tk.Tk()
root.title("Weather App")

tk.Label(root, text="Enter City:").pack()
city_entry = tk.Entry(root)
city_entry.pack()

tk.Button(root, text="Get Weather", command=get_weather).pack(pady=5)
result_label = tk.Label(root, text="")
result_label.pack()

root.mainloop()
